7#define DEBUG 0



8



9void uORB_write_initialize(orb_metadata_t* orbData,



10                          orb_advert_t* orbAdvertObj,



11                          void* busData,



12                          int queueLen) {



13    *orbAdvertObj = orb_advertise_queue(orbData, busData, queueLen);    



14#if DEBUG



15    PX4_INFO("Started advertising topic: %s \n",  orbData->o_name);



16#endif    



17}



18



19void uORB_write_step(orb_metadata_t* orbData,



20                    orb_advert_t* orbAdvertObj,



21                    void* busData) {



22    orb_publish(orbData, *orbAdvertObj, busData);



23}



24



25void uORB_write_terminate(const orb_advert_t* orbAdvertObj) {



26    orb_unadvertise(*orbAdvertObj);



27}
