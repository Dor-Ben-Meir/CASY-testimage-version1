168unsigned pwm_open(int* pwmHandler, 



169                  const char* pwmDevice, 



170                  orb_advert_t* actuatorObj,



171                  orb_advert_t* armObj) {



172#if (defined(PX4_CONNECTEDIO) && (DEBUG == 1))



173    PX4_INFO("* pwm_open: PWM Device: %s * ", pwmDevice);



174    PX4_INFO("*     actuatorObj: %d * ", actuatorObj);



175    PX4_INFO("*     armObj: %d *\n", armObj);



176#endif



177#if defined(MW_PX4_NUTTX_BUILD)    



178    int fd = px4_open(pwmDevice, 0);



179



180    if (fd < 0) {



181        PX4_ERR("can't open %s", pwmDevice);



182        return PWM_OPEN_FAIL;



183    }



184    *pwmHandler = fd;



185



186    PX4_INFO("* Opened PWM Device: %s (fd = %d)*\n", pwmDevice, *pwmHandler);



187#else



188    *pwmHandler = 0;



189	/* advertise actuator_outputs topic */



190	struct actuator_outputs_s outputs;



191	memset(&outputs, 0, sizeof(outputs));



192	*actuatorObj = orb_advertise_queue(ORB_ID(actuator_outputs), &outputs, 1);    



193



194	/* advertise actuator_armed topic */



195	struct actuator_armed_s armPWM;



196	memset(&armPWM, 0, sizeof(armPWM));



197	*armObj = orb_advertise_queue(ORB_ID(actuator_armed), &armPWM, 1);       



198#endif



199    return PWM_SUCCESS;



200}

229unsigned pwm_arm(const int* pwmHandler,



230                 orb_advert_t* armObj) {



231#if (defined(PX4_CONNECTEDIO) && (DEBUG == 1))



232    PX4_INFO("* pwm_arm: *pwmHandler (fd = %d)* ", *pwmHandler);



233    PX4_INFO("*     armObj: %d *\n", armObj);



234#endif



235    /* tell safety that its ok to disable it with the switch */



236#if defined(MW_PX4_NUTTX_BUILD)       



237    int fd = *pwmHandler;



238    int ret = px4_ioctl(fd, PWM_SERVO_SET_ARM_OK, 0);



239



240    if (ret != OK) {



241#if DEBUG



242        err(1, "PWM_SERVO_SET_ARM_OK");



243#endif



244    }



245



246    /* tell IO that the system is armed (it will output values if safety is off) */



247    ret = px4_ioctl(fd, PWM_SERVO_ARM, 0);



248



249    if (ret != OK) {



250#if DEBUG



251        err(1, "PWM_SERVO_ARM");



252#endif



253        return PWM_ARM_FAIL;



254    }



255#else



256	struct actuator_armed_s armPWM;



257	memset(&armPWM, 0, sizeof(armPWM));



258    armPWM.armed = true;



259    orb_publish(ORB_ID(actuator_armed), *armObj, &armPWM);



260#endif



261    return PWM_SUCCESS;



262}

301    int fd = *pwmHandler;

315#endif



316    }



317#else



318	extern int rate;



319#endif



320#if DEBUG



321    PX4_INFO("Frequency %d:", rate);



322#endif



323



324    int ret = px4_ioctl(fd, PWM_SERVO_SET_UPDATE_RATE, rate);



325



326    if (ret != OK) {



327



328#if DEBUG



329        PX4_ERR("PWM_SERVO_SET_UPDATE_RATE (check rate for sanity)");



330#endif



331        return PWM_SETRATE_FAIL;



332    }



333#else



334



335#endif    



336



337    return PWM_SUCCESS;



338}

669    {



670#ifndef PX4_CONNECTEDIO	



671		int pwmValue[8];



672		getDAPWMValues(true, pwmValue);



673#else



674		extern int pwmValue[];



675#endif



676#if (defined(PX4_CONNECTEDIO) && (DEBUG == 1))    



677        uint32_T iPWM;



678        printf("pwm_resetServo: PWM values :");



679        for (iPWM= 0;iPWM<8;iPWM++)



680        {



681            printf("[%d]",pwmValue[iPWM]);



682        }



683        printf("\n");



684#endif        



685        struct actuator_outputs_s outputs;



686        memset(&outputs, 0, sizeof(outputs));



687        outputs.noutputs = servo_count;



688        outputs.timestamp = hrt_absolute_time();



689    for (unsigned i = 0; i < servo_count; i++) {



690            if (channelMask & 1 << i) {



691            outputs.output[i] =  pwmValue[i]; 



692            }



693        }



694        orb_publish(ORB_ID(actuator_outputs), *actuatorObj, &outputs);



695    }



696



697#endif



698    return PWM_SUCCESS;



699



700}

703#if (defined(PX4_CONNECTEDIO) && (DEBUG == 1))

707#endif



708#if defined(MW_PX4_NUTTX_BUILD)      



709    int fd = *pwmHandler;

712    px4_close(fd);



713#else

715    orb_unadvertise(*armObj);



716#endif



717    return PWM_SUCCESS;
